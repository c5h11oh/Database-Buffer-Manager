97 void BufMgr::allocBuf(FrameId & frame)
98 {
99  // check if all of the buffers are pinned
100  bool allIsPinned = true;
101  for(uint i = 0; i < numBufs; ++i){
102  if (bufDescTable[i].pinCnt == 0){
103  allIsPinned = false;
104  break;
105  }
106  }
107  //BufferExceededException Thrown if all buffer frames are pinned
108  if(allIsPinned) {
109  throw BufferExceededException();
110  return;
111  }
112 
113  while(true){
114  advanceClock();
115  // if valid bit is not set, found frame
116  if(!bufDescTable[clockHand].valid){
117  frame = clockHand;
118  return;
119  }
120 
121 
122  // if refbit is set, unset it and restart the loop
123  if(bufDescTable[clockHand].refbit){
124  bufDescTable[clockHand].refbit = false;
125  continue;
126  }
127 
128  // if page is pinned (pinCnt != 0), restart the loop. else we have found the frame
129  if(bufDescTable[clockHand].pinCnt){
130  continue;
131  }
132 
133  // if dirty, we need to write back data first
134  if(bufDescTable[clockHand].dirty){
135  bufDescTable[clockHand].file->writePage(bufPool[clockHand]);
136  }
137 
138  // remove old hash table entry
139  hashTable->remove(bufDescTable[clockHand].file, bufDescTable[clockHand].pageNo);
140 
141  // Clear() the buffer description
142  bufDescTable[clockHand].Clear();
143 
144  // return frame
145  frame = clockHand;
146  return; // It's the caller's (e.g. readPage()'s) responsibility to insert new hashTable entry and call Set()
147  }
148 }

161 {

176  bufPool[returnValue] = file->readPage(pageNo);
177  //Insert the page into the hashtable

179  //Invoke Set() on the frame to set it up properly

181  //Return a pointer to the frame containing the page via the page parameter
182  page = &(bufPool[returnValue]);
